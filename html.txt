        function glowAnimation() {
            var glowText = document.querySelector('.glow-text');
            var level = 0;
            var up = true;
            setInterval(function () {
                glowText.style.textShadow = `0 0 ${10 + level}px rgba(181, 141, 215, 0.35), 0 0 ${20 + level}px rgba(181, 141, 215, 0.35), 0 0 ${30 + level}px rgba(181, 141, 215, 0.35)`;
                if (up) {
                    level += 3;
                    if (level >= 20) {
                        up = false;
                    }
                } else {
                    level -= 3;
                    if (level <= 0) {
                        up = true;
                    }
                }
            }, 100);
        }
        glowAnimation();
